#include <tunables/global>

profile cleanuparr flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Signal handling
  signal (send) set=(kill,term,int,hup,cont),

  # Basic capabilities
  capability,
  file,

  # Network access (required for web UI and API connections)
  network tcp,
  network udp,

  # Home Assistant specific paths
  /data/** rw,
  /config/** rw,

  # Process management (s6-overlay)
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/local/bin/** ix,

  # Temporary files
  /tmp/** rw,
  /run/** rw,

  # Allow access to configuration
  /etc/** r,

  # Deny access to sensitive areas
  deny /var/lib/** rw,
  deny /root/** rw,
}
